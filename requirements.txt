#
# This is safe because parler-tts's actual code works correctly on
# transformers 4.57.x — only its metadata is wrong.
# After installing with --no-deps, verify: python -c "import transformers; print(transformers.__version__)"
# should show >=4.51, NOT 4.46.x.
#
# DO NOT use: pip install parler-tts            (will downgrade transformers)
# DO NOT use: pip install -r requirements.txt   (same problem — use setup.sh instead)
#
